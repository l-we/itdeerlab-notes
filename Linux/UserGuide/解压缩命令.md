
 - tar.xz结尾的压缩文件

```
xz -d *.tar.xz
tar -xvf *.tar
```


 - tar.bz2结尾的压缩文件

```
[root@study root]# yum install bzip2 -y
[root@study root]# bzip2 -d xxx.tar.bz2
[root@study root]# tar -xvf xxx.tar
```


rar格式文件解压缩

```
[root@study root]# wget http://www.rarlab.com/rar/rarlinux-x64-5.3.0.tar.gz
[root@study root]# tar -zxf rarlinux-x64-5.3.0.tar.gz
[root@study root]# cd rar
[root@study root]# make
[root@study root]# make install

```
